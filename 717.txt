class Solution {
public:
    bool isOneBitCharacter(vector<int>& nums) {
        int n=nums.size();
        for(int i=0;i<n-1;i++){
            if(nums[i]==1 && (nums[i+1]==0 || nums[i+1]==1)){
                nums[i] = 2;
                nums[i+1] = 2;
            }
            else if(i==0){
                nums[i]=3;
            }
        }
      
        return nums[n-1]==0;
    }
};
